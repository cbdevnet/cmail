# Prerequisite: Install the required libraries

USER SETUP
1. Clone the repo
2. Run make
3. Run make install
4. Run make tls-init
4a. (Optional) Run make install-doc to install the documentation man pages
5. Edit the installed sample configuration files (default is /etc/cmail)
6. Configure your firewall
7. Run the modules
8. Configure the system via the cmail-admin tools
9. Test the setup

DEVELOPMENT SETUP
(root)	adduser cmail		#create new user
(root)	passwd -l cmail		#no login for this user
(root)	su cmail
(cmail)	git clone <URL>		#clone the repo
(cmail) cd cmail
(cmail)	make			#build the system
# Set up the initial master database
# Set up TLS keys & certificates if needed
# Set up config files
# Set up your firewall to allow connections to the configured ports
(root) /home/cmail/cmail/bin/$MODULE $CONFFILE	#Run the modules
